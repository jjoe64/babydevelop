Quellcode-Strukturen *.cpp-Datei (Beispiel)

Parameter (rot) können verändert werden. 
...
/* <UFGdesigner> */
   this->setGeometry(200,200,159,275);

   label1 = new QLabel("label1",this);
   label1->setPaletteBackgroundColor(yellow);   <-diese Zeile kann gelöscht werden
   label1->setGeometry(10,10,140,50);
   label1->show();

   lineEdit2 = new QLineEdit("lineEdit2",this);
   lineEdit2->setGeometry(10,70,140,30);
   lineEdit2->show();

   pushButton3 = new QPushButton("pushButton3",this);
   pushButton3->setGeometry(10,110,140,40);
   pushButton3->show();

   led4 = new BLed("led4",this);
   led4->setGeometry(40,170,150,90);
   led4->show();

/* </UFGdesigner> */
